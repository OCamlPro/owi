open Syntax

let validate filename =
  let* modul = Parse.Module.from_file filename in
  let+ _modul = Compile.until_typecheck ~unsafe:false modul in
  ()

let cmd debug files =
  if debug then Log.debug_on := true;
  list_iter validate files
