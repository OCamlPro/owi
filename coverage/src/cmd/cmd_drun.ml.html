open Syntax

let cmd ~source_file ~interactive ~unsafe =
  let link_state = Link.State.empty () in
  let+ m, link_state =
    Compile.File.until_link ~unsafe ~name:None link_state source_file
  in
  Denot_interpreter.run ~interactive link_state m
