open Syntax

let validate filename =
  let+ (_modul : Binary.Module.t) =
    Compile.File.until_binary_validate ~unsafe:false ~rac:false ~srac:false
      filename
  in
  ()

let cmd ~debug ~files =
  if debug then Log.debug_on := true;
  list_iter validate files
