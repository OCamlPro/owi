type t = {
  1. id : int;
    (*

    A monotonically increasing id

    *)
  2. kind : kind;
    (*

    Performance or Energy Efficient

    *)
  3. smt : int;
    (*

    The smt/thread id

    *)
  4. core : int;
    (*

    The core id, a core can have multiple smt/threads

    *)
  5. socket : int;
    (*

    The socked id, a socket can have multiple cores

    *)
}
