Module type Memory_sig.COMPLETE_DOMAIN

This is a fully instantiated version of WHOLE_MEMORY_DOMAIN. It is used to break the recursion of the stack of recursive domains (containing uninitiated functors) in modules Codex_register and Dba2Codex

module Scalar : Sig.BASE
module Offset : OFFSET with module Scalar = Scalar
module Value : FIXED_SIZE_VALUE_DOMAIN with module Scalar = Scalar
module Block : BLOCK with module Scalar = Scalar and module Offset = Offset and module Value = Value
module Memory : MEMORY with module Scalar = Scalar and module Address := Value and module Block := Block
